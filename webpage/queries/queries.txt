Section1|Query1|List Employees|"SELECT DISTINCT emp_id AS 'EmployeeID', first_name AS 'First Name', last_name AS 'Last Name' FROM employees ORDER BY RANDOM() LIMIT 30;"
Section1|Query2|List Job Titles|"SELECT DISTINCT title AS 'Job Title' FROM titles ORDER BY RANDOM();"
Section1|Query3|List Employees Ordered By Name|"SELECT DISTINCT first_name AS 'First Name',  last_name AS 'Last Name' FROM employees ORDER BY first_name, last_name ASC LIMIT 10;"
Section2|Query1|List Department Titles|"SELECT DISTINCT dept_name AS 'Department Title' FROM departments;"
Section2|Query2|List Total Number of Employees|"SELECT COUNT(DISTINCT emp_id) AS 'N Employees' FROM employees;"
Section2|Query3|List Female Employees|"SELECT * FROM employees WHERE gender = 'F' GROUP BY emp_ID ORDER BY RANDOM() LIMIT 30;"
Section3|Query1|List Employees Working Since May-1991|"SELECT COUNT(DISTINCT emp_id) AS 'Employee Count' FROM employees WHERE hire_date = '1991-05-01';"
Section3|Query2|List Count of Postitions Held Per Employees|"SELECT emp_id, COUNT(DISTINCT title) AS 'N_Titles' FROM titles GROUP BY emp_id HAVING N_Titles >= 2 ORDER BY N_Titles LIMIT 10;"
Section3|Query3|Count Active Salaries Between 90,000 - 90,004|"SELECT COUNT(DISTINCT emp_id) AS 'Employee Count' FROM salaries WHERE to_date = '9999-01-01' AND salary BETWEEN(90000 AND 90004);"
